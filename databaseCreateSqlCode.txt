CREATE DATABASE electronics;
\c electronics;
CREATE TABLE providers (
    name VARCHAR(100) PRIMARY KEY,
    address VARCHAR(200) NOT NULL,
    phone VARCHAR(20) NOT NULL CHECK(phone ~ '\+7[0-9]{10}'),
    inn CHAR(10) NOT NULL UNIQUE
);

CREATE TABLE products (
    name VARCHAR(100) PRIMARY KEY,
    provider VARCHAR(100) NOT NULL REFERENCES providers(name) ON UPDATE CASCADE ON DELETE RESTRICT,
    category VARCHAR(50) NOT NULL,
    price DECIMAL(12, 2) NOT NULL CHECK (price > 0),
    count INT NOT NULL CHECK (count >= 0) DEFAULT 0
);

CREATE TABLE employees (
    id SERIAL PRIMARY KEY,
    fio VARCHAR(100) NOT NULL,
    position VARCHAR(50) NOT NULL,
    contact_data VARCHAR(100) NOT NULL,
    admission_date DATE NOT NULL DEFAULT CURRENT_DATE,
    dismissal_date DATE DEFAULT NULL CHECK(dismissal_date IS NULL OR dismissal_date > admission_date)
);

CREATE TABLE clients (
    id SERIAL PRIMARY KEY,
    fio VARCHAR(100) NOT NULL,
    contact_data VARCHAR(100) NOT NULL
);

CREATE TABLE orders (
    id SERIAL PRIMARY KEY,
    client_id INT NOT NULL REFERENCES clients(id) ON UPDATE CASCADE ON DELETE RESTRICT,
    employee_id INT NOT NULL REFERENCES employees(id) ON UPDATE CASCADE ON DELETE RESTRICT,
    total_price DECIMAL(12, 2) NOT NULL DEFAULT 0,
    date TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE order_details (
    id SERIAL PRIMARY KEY,
    order_id INT NOT NULL REFERENCES orders(id) ON UPDATE CASCADE ON DELETE CASCADE,
    product_name VARCHAR(100) NOT NULL REFERENCES products(name) ON UPDATE CASCADE ON DELETE RESTRICT,
    count INT NOT NULL CHECK (count > 0),
    price DECIMAL(12, 2) NOT NULL DEFAULT 0
);

-- 1. Функция для вычисления price в order_details
CREATE OR REPLACE FUNCTION set_order_detail_price()
RETURNS TRIGGER AS $$
BEGIN
    NEW.price := (
        SELECT price FROM products WHERE name = NEW.product_name
    ) * NEW.count;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- 2. Функция для обновления total_price
CREATE OR REPLACE FUNCTION update_order_total_price()
RETURNS TRIGGER AS $$
BEGIN
    UPDATE orders 
    SET total_price = (
        SELECT COALESCE(SUM(price), 0)
        FROM order_details 
        WHERE order_id = COALESCE(NEW.order_id, OLD.order_id)
    )
    WHERE id = COALESCE(NEW.order_id, OLD.order_id);
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

-- 3. Функция проверки наличия товара
CREATE OR REPLACE FUNCTION check_product_availability()
RETURNS TRIGGER AS $$
DECLARE
    available_count INT;
BEGIN
    SELECT count INTO available_count
    FROM products WHERE name = NEW.product_name;
    
    IF available_count < NEW.count THEN
        RAISE EXCEPTION 'Недостаточно товара "%". Доступно: %, Заказано: %', 
            NEW.product_name, available_count, NEW.count;
    END IF;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- 4. Функция обновления склада
CREATE OR REPLACE FUNCTION update_product_stock()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'INSERT' THEN
        UPDATE products SET count = count - NEW.count WHERE name = NEW.product_name;
    ELSIF TG_OP = 'UPDATE' THEN
        UPDATE products SET count = count + OLD.count - NEW.count WHERE name = NEW.product_name;
    ELSIF TG_OP = 'DELETE' THEN
        UPDATE products SET count = count + OLD.count WHERE name = OLD.product_name;
    END IF;
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

-- Создаем триггеры
CREATE TRIGGER trg_set_order_detail_price
BEFORE INSERT OR UPDATE ON order_details
FOR EACH ROW EXECUTE FUNCTION set_order_detail_price();

CREATE TRIGGER trg_update_order_total
AFTER INSERT OR UPDATE OR DELETE ON order_details
FOR EACH ROW EXECUTE FUNCTION update_order_total_price();

CREATE TRIGGER trg_check_product_availability
BEFORE INSERT OR UPDATE ON order_details
FOR EACH ROW EXECUTE FUNCTION check_product_availability();

CREATE TRIGGER trg_update_product_stock
AFTER INSERT OR UPDATE OR DELETE ON order_details
FOR EACH ROW EXECUTE FUNCTION update_product_stock();
